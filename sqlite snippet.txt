using System.Data.SQLite;

SQLiteConnection.CreateFile("TestDatabase.sqlite");

var conn = new SQLiteConnection("Data Source=TestDatabase.sqlite;Version=3;");
conn.Open();

string sql = "create table highscores (name varchar(20), score int)";
var comm = new SQLiteCommand(sql, conn);
comm.ExecuteNonQuery();

sql = "insert into highscores (name, score) values ('Me', 3000)";
comm = new SQLiteCommand(sql, conn);
comm.ExecuteNonQuery();

sql = "select * from highscores order by score desc";
comm = new SQLiteCommand(sql, conn);
var reader = comm.ExecuteReader();

while (reader.Read())
{
    var name = reader["name"];
    var score = reader["score"];
}

conn.Close();
